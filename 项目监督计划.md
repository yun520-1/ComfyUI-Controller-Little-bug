## 🔧 技术实现

### 搜索脚本

```python
#!/usr/bin/env python3
"""
ComfyUI 项目监督脚本
- GitHub API 搜索
- ClawHub API 搜索
- 项目评估
- 自动下载
"""

import requests
import json
from datetime import datetime, timedelta

def search_github(keywords, min_stars=50, max_age_days=90):
    """GitHub 搜索"""
    results = []
    for keyword in keywords:
        query = f"{keyword} stars:>{min_stars} pushed:>{(datetime.now() - timedelta(days=max_age_days)).strftime('%Y-%m-%d')}"
        response = requests.get(
            f"https://api.github.com/search/repositories?q={query}&sort=stars&order=desc"
        )
        if response.status_code == 200:
            data = response.json()
            results.extend(data['items'][:10])
    return results

def evaluate_project(repo):
    """项目评估"""
    score = 0
    
    # Stars 评分 (40%)
    stars = repo.get('stargazers_count', 0)
    if stars >= 100:
        score += 40
    elif stars >= 50:
        score += 30
    
    # 更新时间评分 (30%)
    updated_at = datetime.strptime(repo['updated_at'], '%Y-%m-%dT%H:%M:%SZ')
    days_since_update = (datetime.now() - updated_at).days
    if days_since_update <= 30:
        score += 30
    elif days_since_update <= 90:
        score += 20
    
    # 功能独特性 (20%)
    # 需要人工评估
    
    # 代码质量 (10%)
    if repo.get('has_wiki') and repo.get('has_issues'):
        score += 10
    
    return score
```
